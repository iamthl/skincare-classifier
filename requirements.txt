# --- Runtime ---------------------------------------------------------------

# --- Static typing ---------------------------------------------------------

# Bandit performs static analysis against common Python security pitfalls.
bandit==1.7.10
# pip-audit cross-references installed packages against the PyPA
# vulnerability database (OSV / GHSA) - the SCA control referenced
# in the Week 11 lecture.
pip-audit==2.7.3
